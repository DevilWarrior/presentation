  .Does(() =>
  {
    Information("Hello from CAKE");
  });

RunTarget(target);

Task("Default")
  .IsDependentOn("Initialization");

Task("Initialisation");

CAKE


Task("Default")
  .Does(() =>
{
    Information("Hello from CAKE");
    Just.Say("Something");
});

class Just
{
    public static void Say (string str)
    {
        Information(str);
    }
}

CAKE


Task("Default")
  .Does(() =>
{
    Information("Hello from CAKE");
    Just.Say("Something", Context);
});

class Just
{
    public static void Say (string str, ICakeContext ctx)
    {
        ctx.Information(str);
    }
}
